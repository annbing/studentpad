function liveCards(){ return [...document.querySelectorAll('.swipe')].filter(c=>!c.dataset.gone); }

function cardTone(btn,tone,e){ if(e) e.stopPropagation(); const card=btn.closest('.swipe'); card.querySelectorAll('.tc').forEach(t=>t.setAttribute('aria-pressed','false')); btn.setAttribute('aria-pressed','true'); card.querySelector('[data-draft]').textContent=OPPS[card.dataset.id].drafts[tone]; }

let drag=null;

function attachTop(){

  const card=topCardEl(); if(!card){ showEmpty(); return; }

  card.onpointerdown=e=>{ if(e.target.closest('.no-drag')) return; drag={card,x0:e.clientX,y0:e.clientY,dx:0}; card.setPointerCapture(e.pointerId); card.style.transition='none'; };

  card.onpointermove=e=>{ if(!drag) return; drag.dx=e.clientX-drag.x0; card.style.transform=`translateX(${drag.dx}px) rotate(${drag.dx/18}deg)`; card.querySelector('.stamp.meet').style.opacity=Math.max(0,drag.dx/120); card.querySelector('.stamp.later').style.opacity=Math.max(0,-drag.dx/120); };

  card.onpointerup=()=>{ if(!drag) return; const dx=drag.dx; card.style.transition='transform .35s ease'; if(Math.abs(dx)>110){ resolve(card, dx>0?1:-1); } else { card.style.transform=''; card.querySelectorAll('.stamp').forEach(s=>s.style.opacity=0); } drag=null; };

}

function flick(dir){ const card=topCardEl(); if(!card){ return; } card.style.transition='transform .4s ease'; resolve(card,dir); }

function resolve(card,dir){

  const id=card.dataset.id;

  card.querySelector(dir>0?'.stamp.meet':'.stamp.later').style.opacity=1;

  card.style.transform=`translateX(${dir*600}px) rotate(${dir*28}deg)`;

  card.dataset.gone='1'; history.push(id);

  setTimeout(()=>{ card.style.display='none'; if(dir>0){ openMessage(id); } else { announce('Set aside'); attachTop(); if(!liveCards().length) showEmpty(); } },330);

}

function undo(){

  if(!card) return;

  $('deckbtns').style.display='flex';

function renderHeatmap(){

  // Generate 12 weeks of reconnection data (actual counts)

  const reconnections=[0,1,2,1,4,3,5,2,3,1,2,6];

  const weekDates = [];

  const today = new Date();

 

  // Generate week labels (going backwards from today)

  for(let i=11; i>=0; i--){

    const weekStart = new Date(today);

    weekStart.setDate(today.getDate() - (i * 7));

    const month = weekStart.toLocaleDateString('en-US', {month:'short'});

    const day = weekStart.getDate();

    weekDates.push(`${month} ${day}`);

  }

 

  $('heatmapWeeks').innerHTML = reconnections.map((count,i)=>{

    let level = 0;

    if(count >= 7) level = 4;

    else if(count >= 5) level = 3;

    else if(count >= 3) level = 2;

    else if(count >= 1) level = 1;

   

    const label = count === 0 ? 'No reconnections' : count === 1 ? '1 reconnection' : `${count} reconnections`;

    return `<div class="hm-square${level>0?' l'+level:''}" title="Week of ${weekDates[i]}: ${label}" role="img" aria-label="Week of ${weekDates[i]}: ${label}"></div>`;

  }).join('');

}
